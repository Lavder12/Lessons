create table Person (
personId int primary key,
lastName varchar (64),
firstName varchar(64)
)

CREATE TABLE Address (
    addressId INT,
    personId INT,
    city VARCHAR(100),
    street VARCHAR(100),
    PRIMARY KEY (addressId, personId), -- Составной первичный ключ
    FOREIGN KEY (personId) REFERENCES Person(personId)
);
